// Shared WebAuthn utility functions — included inline by register.html and login.html. function b64urlToBuffer(b64url) { const b64 = b64url.replace(/-/g, '+').replace(/_/g, '/'); const binary = atob(b64); const buf = new ArrayBuffer(binary.length); const bytes = new Uint8Array(buf); for (let i = 0; i < binary.length; i++) bytes[i] = binary.charCodeAt(i); return buf; } function bufferToB64url(buffer) { const bytes = new Uint8Array(buffer); let binary = ''; bytes.forEach(b => (binary += String.fromCharCode(b))); return btoa(binary).replace(/\+/g, '-').replace(/\//g, '_').replace(/=/g, ''); } function getCsrf() { return document.querySelector('meta[name="csrf-token"]').content; } function setStatus(el, msg) { el.textContent = msg; el.hidden = !msg; }
